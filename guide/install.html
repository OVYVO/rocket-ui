适用于 Vue2.x 的简易组件库

# 安装

npm install rockets-ui -S
// or
yarn add rockets-ui -S
// or
cnpm install rockets-ui -S

# 快速开始

import Vue from 'vue'
import RocketsUI from 'rockets-ui'
import 'rockets-ui/lib/index.css'

Vue.use(RocketsUI)

// or
import {
  Statistics
  // ...
} from 'rockets-ui'
import 'rockets-ui/lib/index.css'

Vue.component(Statistics.name, Statistics)
