适用于Vue2.0的简易组件库

# 安装

npm install rocket-ui -S
// or
yarn add rocket-ui -S
// or
cnpm install rocket-ui -S

# 快速开始

import Vue from 'vue'
import RocketUI from 'rocket-ui'
import 'rocket-ui/dist/index.css'

Vue.use(RocketUI)

// or
import {
  Statistics
  // ...
} from 'rocket-ui'
import 'rocket-ui/dist/index.css'

Vue.component(Statistics.name, Statistics)
